declare -> LET ID@name COLON type@type {
    name.type: "variable",
    name.modifier: "declare",
    type.scope: scope,
};

declare -> LET ID@name ASSIGN expr@init {
    name.type: "variable",
    name.modifier: "declare",
    type: init.type
};

func -> FUNC ID@name LP ...params RP ...block {
    stmts.scope: scope + name
};

block -> LS ...stmts RS;

stmts -> stmt@stmts SEMI
    | ...stmts stmt@stmts SEMI;

stmt -> ....;
